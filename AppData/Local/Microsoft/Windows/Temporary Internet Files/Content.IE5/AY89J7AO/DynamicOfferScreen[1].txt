            function getElementsByClassName(searchClass, node, tag) {
                var classElements = new Array();
                if (node == null)
                    node = document;
                if (tag == null)
                    tag = '*';
                var els = node.getElementsByTagName(tag);
                var elsLen = els.length;
                var pattern = new RegExp("(^|\\s)" + searchClass + "(\\s|$)");
                for (i = 0, j = 0; i < elsLen; i++) {
                    if (pattern.test(els[i].className)) {
                        classElements[j] = els[i];
                        j++;
                    }
                }
                return classElements;
            }

                            counter++;
                        }
                        else {
                            if (isfirst) {
                                isfirst = false;
                                rejected += pro8label.innerText;
                            }
                            else {
                                rejected += ", " + pro8label.innerText;
                            }
                        }
                    }

                    if (pro9label.innerText != "") {
                        if (pro9.checked == true) {
                            cmd += "VirtualOfferCmd:()-";
                            counter++;
                        }
                        else {
                            if (isfirst) {
                                isfirst = false;
                                rejected += pro9label.innerText;
                            }
                            else {
                                rejected += ", " + pro9label.innerText;
                            }
                        }
                    }

                    if (pro10label.innerText != "") {
                        if (pro10.checked == true) {
                            cmd += "VirtualOfferCmd:()-";
                            counter++;
                        }
                        else {
                            if (isfirst) {
                                isfirst = false;
                                rejected += pro10label.innerText;
                            }
                            else {
                                rejected += ", " + pro10label.innerText;
                            }
                        }
                    }
                }

                if (quick.checked == true && quick.disabled == false || counter == productCounter()) {

                    allProducts = true;
                }
                else {
                    allProducts = false;
                }

                return cmd;
            }

            function OnPopupCancel() {
                if (cancelbuttoncounter == 0) {
                    document.getElementById("cancelbuttonlabel").style.color = "gray";
                    document.getElementById("onokbuttonlabel").style.color = "gray";
                    document.getElementById("cancelbutton").disabled = true;
                    document.getElementById("onokbutton").disabled = true;
                    document.getElementById("idAccept").disabled = true;
                    document.getElementById("idAccept").style.color = "gray";

                    elm.style.visibility = "hidden";
                    elm.style.display = "none";
                    if (cmd == "") {
                        OnCancel();
                    }
                    else {
                        cmd = cmd.substring(0, cmd.length);
                        if (getUrlParameterByName('external') == '0') {
                            window.location = "http://onnext?cmd=" + cmd;
                        } else {
                            window.external.OnNext(cmd);
                        }
                    }
                    cancelbuttoncounter++;
                }
            }

            function OnOk() {
                if (okbuttoncounter == 0) {
                    okbuttoncounter++;
                    document.getElementById("cancelbuttonlabel").style.color = "gray";
                    document.getElementById("onokbuttonlabel").style.color = "gray";
                    document.getElementById("cancelbutton").disabled = true;
                    document.getElementById("onokbutton").disabled = true;
                    document.getElementById("idAccept").disabled = true;
                    document.getElementById("idAccept").style.color = "gray";

                    quick.disabled = false;

                    if (cmd == "") {

                    }
                    else {
                        cmd = cmd.substring(0, cmd.length);
                        if (getUrlParameterByName('external') == '0') {
                            window.location = "http://onnext?cmd=" + cmd;
                        } else {
                            window.external.OnNext(cmd);
                        }
                    }
                }

            }

                                    window.location = "http://onnext?cmd=" + cmd;
                                } else {
